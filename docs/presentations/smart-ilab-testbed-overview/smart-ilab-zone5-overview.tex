\begin{frame}{Staging Compose Vs. Production systemd}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Production}
\begin{itemize}
  \item Branch: \code{main}
  \item Runtime: user-level systemd
  \item Ports: backend \code{8000}, Vite \code{8015}
  \item Source deploy refuses active trainer collisions.
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{Staging}
\begin{itemize}
  \item Branch: \code{cicd/zone5-compose}
  \item Runtime: Docker Compose
  \item Ports: backend \code{8005}, Vite \code{8016}
  \item Copies env and model pointer from production when needed.
\end{itemize}
\end{block}
\end{columns}
\tightsep
\begin{center}
\large\textbf{Source deploy, staging deploy, training, and model delivery stay separate.}
\end{center}
\end{frame}

\sectionframe{04}{Labels And Methodology}

\begin{frame}{Zone 5 Labels}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\textbf{CV truth}
\begin{itemize}
  \item One camera, one Desk 5 mask, one 10-second Zone 5 label stream.
  \item Target is \code{zone_occupied}.
  \item Occupied when median person count crosses the configured threshold.
\end{itemize}

\column{0.48\textwidth}
\textbf{Guardrails}
\begin{itemize}
  \item Missing labels stay null until a real CV bucket exists.
  \item Sensor missingness is diagnostic metadata, not an occupancy signal.
  \item Trainer snapshots the rolling CSV before fitting.
\end{itemize}
\end{columns}
\tightsep

\begin{frame}[shrink=3]{Zone 5 Methodology}
\begin{columns}[T,onlytextwidth]
\column{0.46\textwidth}
\textbf{Feature families}
\begin{itemize}
  \item AIR-1 Zone 5: temperature, humidity, CO2, PM2.5.
  \item Smart plug: \code{power_s5}.
  \item mmWave: \code{mmwave_s5} and recency context.
  \item SEN55: particulate, environment, VOC, NOx.
  \item Time: hour and day-of-week cycles.
\end{itemize}

\column{0.50\textwidth}
\begin{block}{Training contract}
\begin{itemize}
  \item Current marker: \code{zone5_mmwave_recency_v1}.
  \item Missing indicators are diagnostics and metadata, not model inputs.
  \item Legacy \code{zone5_missingness_decoupled_v1} artifacts remain loadable for live compatibility.
  \item Core live gating covers AIR-1, \code{power_s5}, and \code{mmwave_s5}; SEN55 is optional.
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Verification And Roadmap}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\textbf{Verification gates}
\begin{itemize}
  \item Python compile checks for package and app modules.
  \item Zone 5 ground-truth contract tests.
  \item Vite build checks.
  \item Systemd unit verification.
  \item Docker Compose config and image build.
  \item \code{/api/health} confirms app health and model readiness fields.
\end{itemize}

\column{0.48\textwidth}
\textbf{Roadmap and risks}
\begin{itemize}
  \item Prove manual Zone 5 retraining before enabling scheduled retrain.
  \item Add model freshness, promotion, and upstream API latency monitoring.
  \item Live QA depends on lab network, API, cameras, and devices.
  \item Training quality depends on both-class CV labels across usable windows.
\end{itemize}
\end{columns}
\end{frame}
