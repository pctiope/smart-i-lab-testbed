\column{0.48\textwidth}
\textbf{Operational view}
\begin{itemize}
  \item One shared AIR1 model reports per-zone probabilities for zones 1--15.
  \item Two camera streams feed separate zone masks and one per-zone label table.
  \item The app can remain up while model readiness is blocked by a missing production pointer.
\end{itemize}

\textbf{Important separation}
\begin{itemize}
  \item AIR1 excludes power and mmWave from model inputs.
  \item Missing camera coverage remains null; it is not converted to vacant.

\sectionframe{03}{Labels And Pipeline}

\begin{frame}{AIR1 Labels}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\textbf{CV truth}
\begin{itemize}
  \item Two cameras, per-zone masks, and one label table keyed by \code{timestamp + zone_id}.
  \item Target is \code{occupied}.
  \item Occupied when per-zone median \code{occupancy_count >= 1}.
\end{itemize}

\column{0.48\textwidth}
\textbf{Null handling}
\begin{itemize}
  \item A row with no matching CV label keeps \code{occupied = null}.
  \item Missing camera coverage is not converted into a negative label.
  \item Table 16 is excluded from the model-zone contract.
\end{itemize}
\end{columns}
\tightsep

\begin{frame}[shrink=3]{All-Zones Feature Contract}
\begin{columns}[T,onlytextwidth]
\column{0.46\textwidth}
\textbf{Input families}
\begin{itemize}
  \item AIR-1: temperature, humidity, CO2, PM2.5.
  \item SEN55: particulate, environment, VOC, NOx.
  \item Missing indicators remain part of the AIR1 feature contract.
  \item Time: hour and day-of-week cycles.
\end{itemize}

\column{0.50\textwidth}
\begin{block}{Contract guards}
\begin{itemize}
  \item Rows are long-form: one \code{timestamp + zone_id} pair per zone.
  \item Valid model zones are \code{1--15}.
  \item \code{zone_id} supports grouping, splits, audits, and display only.
  \item Power and mmWave are excluded.
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\sectionframe{04}{Verification And Roadmap}

\begin{frame}{Verification Gates}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\textbf{Contract checks}
\begin{itemize}
  \item \code{test_all_zones_contract.py} covers features, labels, zone masks, windows, and runtime payloads.
  \item Smoke checks reject candidates with the wrong target column.
  \item Package docs preserve the no-power/no-mmWave AIR1 boundary.
\end{itemize}

\column{0.48\textwidth}
\textbf{Runtime checks}
\begin{itemize}
  \item \code{/api/health} reports readiness, history size, and latest runtime errors.
  \item The dashboard can show camera freshness even when inference is waiting for a promoted model.
  \item Person counters keep writing annotated frames during upstream outages.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Roadmap And Risks}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\textbf{Near-term roadmap}
\begin{itemize}
  \item Promote a production AIR1 all-zones model after sufficient strict-window evidence.
  \item Keep tracker, collector, trainer, and app service health visible.
  \item Add model freshness and upstream API latency monitoring.
\end{itemize}

\column{0.48\textwidth}
\textbf{Risks to keep visible}
\begin{itemize}
  \item Label quality depends on two camera streams and per-zone mask coverage.
  \item Training quality depends on both-class labels across usable calendar windows.
  \item Missing production pointers should degrade inference, not dashboard availability.
\end{itemize}
\end{columns}
\end{frame}
